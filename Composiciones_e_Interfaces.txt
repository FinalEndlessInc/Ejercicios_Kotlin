interface Habilidad{
    var nombre: String
    var activo: Boolean
    var descripcion: String?
    
    fun activar(): Boolean{
        if(activo){
            return false
        }
        else{
            activo = true
            return false
        }
    }
    
    fun desactivar(): Boolean{
        if(!activo){
            return false
        }
        else{
            activo = false
            return true
        }
    }
    
}

interface Habilidades{
    var capacidades: MutableList<Habilidad>

    fun activar_capacidad(nombre:String): Boolean{
        for(capacidad in capacidades){
            if (capacidad.nombre == nombre){
                return capacidad.activar()
            }
        }
        return false
    }
    
    fun agregar_capacidad(capacidad_nueva:Habilidad){
        capacidades.add(capacidad_nueva)
    }
    
    fun descripcion(): String{
        var lista_de_capacidades = "${capacidades.size}: "
        
        for(capacidad in capacidades){
            lista_de_capacidades = "${lista_de_capacidades}, ${capacidad.nombre}"
        }
        
        return lista_de_capacidades
    }
}

interface Pasatiempos{
    var actividad: String
    var descripcion_actividad: String
}

interface Hobbies{
    var hobbies: MutableList<Pasatiempos>
    
    fun agregar_hobbie(nuevo_hobbie: Pasatiempos){
        hobbies.add(nuevo_hobbie)
    }
    
    fun ver_hobbies(): String{
        var lista_de_hobbies = "${hobbies.size}: "
        for(hobbie in hobbies){
            lista_de_hobbies = "${lista_de_hobbies}, ${hobbie.actividad}"
        }
        
        return lista_de_hobbies
    }
}

interface Cumbias{
    var estilo: String
    var bailando: Boolean
    
    fun esta_bailando(estilo: estiloBaile):String{
        if(bailando){
            return "Joer, que wen baile de ${estilo.estilo}"
        }else{
            return "Nah, este bebe no baila"
        }
    }
}

interface Gameplay{
    var nombre_juego: String
    var partidas_jugadas: Int
    var partida_ganada: Boolean
    
    fun ganar_partida(partidas: BebeJugando): String{
        if (partida_ganada){
            return " esta jugando ${partidas.nombre_juego}, ya jugó ${partidas.partidas_jugadas} y ha ganado, yippie !"
        }else{
            return " esta jugando ${partidas.nombre_juego}, ya jugó ${partidas.partidas_jugadas} y perdió, que triste"
        }
    }
}
